# 文档目录结构

- config # 配置，构建等配置
- public
  - favicon.png
  - index.html
- src
  - assets # 本地静态资源
  - components # 业务通用组件
  - layouts # 通用布局
  - models # redux 处理层
  - pages # 业务页面入口和常用模板
  - services # 后台接口服务
  - store # redux 配置
  - utils # 工具库
  - global.less # 全局样式
  - index.ts # 入口
- README.md
- package.json
